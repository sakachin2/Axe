#ifdef UNX
   = CHDir   [pathname | *]
               set to home directory if operand missing.
               * : set from file/directory currently displayed.
#else
   = CHDir   [[d:]pathname | *]
               display current directory if operand missing
               * : set from file/directory currently displayed.
  #ifdef W32
               UNC format(\\server1\\dir1\...) is treated under the psude drive "\:".
               After "CD \\server1\dir1",
               \:file means \\server1\dir1\file, \:\file means \\server1\file.
  #endif
#endif

             wildcard is available if unique path determined.
